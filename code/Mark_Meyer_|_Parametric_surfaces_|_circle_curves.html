from math import sin, cos
def circle_equation(r, dt):
    center_x = 0.0
    center_y = 0.0
    t = 0.0
    while True:
        x = center_x + cos(t) * r
        y = center_y + sin(t) * r
        yield x, y
        t += dt
 
b = 5.5
eq1 = circle_equation(-135, -b)
eq2 = circle_equation(-165, b)
eq3 = circle_equation(150, b)
eq4 = circle_equation(325, b)
x1, y1 = eq1.next()
x2, y2 = eq1.next()
